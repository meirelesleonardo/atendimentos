CREATE TABLE atendimentos (
  id int(11) NOT NULL PRIMARY KEY,

  mat_solicitante VARCHAR(45) DEFAULT NULL,
  unidade VARCHAR(100) DEFAULT NULL,
  sistema VARCHAR(100) DEFAULT NULL,
  sol VARCHAR(50) DEFAULT NULL,
  forma_contato VARCHAR(100) DEFAULT NULL,
  tipo_servico VARCHAR(100) DEFAULT NULL,
  grupo_servico VARCHAR(100) DEFAULT NULL,
  solicitacao_inicio DATETIME DEFAULT NULL,
  solicitacao_fim DATETIME DEFAULT NULL,
  descricao TEXT DEFAULT NULL,
  remoto VARCHAR(200) DEFAULT NULL,
  hInicio TIME DEFAULT NULL,
  hFim TIME DEFAULT NULL,
  ip VARCHAR(45) DEFAULT NULL
);
